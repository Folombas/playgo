### Новые структуры
```go
// Particle — частица для визуальных эффектов
type Particle struct {
    X, Y     float32
    VX, VY   float32
    Life     int
    MaxLife  int
    Color    color.RGBA
    Size     float32
    Gravity  float32
}

// ScreenShake — тряска экрана
type ScreenShake struct {
    Intensity float32
    Duration  int
    Timer     int
    Angle     float64
}
```
